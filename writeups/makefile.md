# ==============================================================================
# 1. COMPILER & FLAGS CONFIGURATION
# What this is : 
# These are just variables : we declare variables in make language via variable := value, and no quotes are involved. To call that variable (ie to access it) we use $(variable).

# Syntax of the make system is deceptively simple
# It is just like build dependencies for a specific target. 
# .PHONY is a special target (which is discussed later.)

# target: dependency1 dependency2 ....
#	instructions for building the target. 
# dependency1: 
# 	instructions for building dependency1. 

# First the dependencies get built, then comes the topmost target. 

# ==============================================================================

# we shall use the gcc compiler
CC       := gcc

# the compiler flags to be used. 
CFLAGS   := -Wall -Wextra -Wpedantic -O2 -std=c23

# the c preprocessor flags to be used. (essentially are also compiler flags.)
CPPFLAGS := -Iinclude -MMD -MP

# the name of our target program : ie the binary. just keep whatever name you have.
TARGET   := my_program

# here are the source and object folders. 
# in the source dir is the source files - and in the obj directory is the obj files. 
SRC_DIR  := src

# our obj directory is dynamically created and destroyed in the make process.
OBJ_DIR  := obj

# ==============================================================================
# 2. DYNAMIC FILE DETECTION (honestly, do some research)
# ==============================================================================

# Find all .c files in the src directory (eg : src/main.c src/utils.c)
# For this we make use of the wildcard (so that we can use *)
# hey : expand SRC_DIR and grab all the files ending with .c in there. 
SRCS     := $(wildcard $(SRC_DIR)/*.c)

# Convert the list of .c files into .o files inside the obj directory
# (eg :  obj/main.o obj/utils.o)
# the % here is telling the make system to keed the name unmodified
OBJS     := $(SRCS:$(SRC_DIR)/%.c=$(OBJ_DIR)/%.o)

# then we generate a list of dependency tracking files (.d) for every object file in obj directory iteself.
DEPS     := $(OBJS:.o=.d)

# ==============================================================================
# 3. BUILD RULES
# ==============================================================================
# .PHONY is a special built-in target used to tell make that a specific target name is not a physical file on the computer. It is fake name for a recipe of commands that we want to run. 
# here we have all and clean : two fake targets. 
.PHONY: all build clean


# for building : we can do either : make build, make all, make clean for cleaning. 
# Default target alias to build 
all: build

build: $(TARGET)

# Rule to link the final executable
# single dependency : objs should exist. 
# We need to build all the obj files before this build is done. 
$(TARGET): $(OBJS)
	# @ will silence it from echoing the command itself. $@ is the target name.
	# here $@ is $(TARGET)
	@echo "Linking executable: $@"
	# gcc ..flags $^ means all dependencies (all objects in $(OBJS)) -o $(TARGET)
	# just read it again and you will understand. 
	$(CC) $(CFLAGS) $^ -o $@

# Pattern rule to compile .c files to .o files
# each object file in obj/... require src/... with same name execept for swapped o with c. 
# but we need the obj directory to exist first. 
# | is called the order only dependency. 
# anything to left of | is the normal dependency. 
# anything to the right is the order only dependency.
# if the obj_dir does not exist, make will create it first. 
# ie, anything in order only dependency will be made only if it does not exist, else left as it is. 
$(OBJ_DIR)/%.o: $(SRC_DIR)/%.c | $(OBJ_DIR)
	@echo "Compiling: $< -> $@"
	# we here have the $< is the first dependency : which is src/target.c
	# our target is obj/target.o : first compilation in the obj directory. 
	# before that obj directory will be created. 
	$(CC) $(CFLAGS) $(CPPFLAGS) -c $< -o $@

# Rule to guarantee the obj/ directory exists
$(OBJ_DIR):
	mkdir -p $(OBJ_DIR)

# Clean rule
clean:
	@echo "Cleaning up..."
	rm -rf $(OBJ_DIR) $(TARGET)
	# if we need some kind of auto clean all obj files, we can add it for all: clean $(TARGET)

# ==============================================================================
# 4. HEADER DEPENDENCY INCLUSION
# ==============================================================================
# Include the dependency files generated by GCC (-MMD -MP flags)
-include $(DEPS)
